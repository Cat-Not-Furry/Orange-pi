# 1. Información del sistema
uname -a > sysinfo.txt
cat /etc/os-release >> sysinfo.txt
lscpu >> sysinfo.txt
free -h >> sysinfo.txt
lsusb >> sysinfo.txt   # para ver dispositivos USB (cámara)
lsmod | grep video >> sysinfo.txt

# 2. Información de la cámara (usando v4l2-ctl)
sudo apt install v4l-utils -y
v4l2-ctl --list-devices > cam_info.txt
v4l2-ctl -d /dev/video0 --all >> cam_info.txt
v4l2-ctl -d /dev/video0 --list-formats-ext >> cam_info.txt

# 3. Información de red
ip a > network.txt
iwconfig >> network.txt  # si es WiFi
iw dev wlan0 station dump >> network.txt  # estadísticas de la estación WiFi
ping -c 10 IP_DE_TU_LAPTOP > ping_test.txt   # prueba de latencia y pérdida

# 4. Log de prueba de transmisión con FFmpeg (captura y envía a UDP)
# Esto ejecutará FFmpeg durante 30 segundos y guardará logs
ffmpeg -f v4l2 -i /dev/video0 -c:v libx264 -preset ultrafast -tune zerolatency -b:v 2000k -f mpegts udp://IP_LAPTOP:1234 -loglevel debug 2> ffmpeg_log.txt
